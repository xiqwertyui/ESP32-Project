void setup(void) {
  const int readyLED = 26;
  Wire.begin();
  pinMode(readyLED, OUTPUT);
  digitalWrite(readyLED, LOW);
  u8g2.begin();
  Serial.begin(9600);
  Serial.println("raptor frames");
  digitalWrite(readyLED, HIGH);
}

uint8_t toggle = 0;

void loop(void) {
  u8g2.firstPage();
  do {
    Serial.print("toggle ");
    Serial.println(toggle);
    Serial.println();
    if (toggle == 00)
      u8g2.drawBitmap(8, 0, 14, 62, dino_018);  // dino 112*62
    else if (toggle == 01)
      u8g2.drawBitmap(8, 0, 14, 62, dino_019);  // dino 112*62
    else if (toggle == 02)
      u8g2.drawBitmap(8, 0, 14, 62, dino_020);  // dino 112*62
    else if (toggle == 03)
      u8g2.drawBitmap(8, 0, 14, 62, dino_021);  // dino 112*62
    else if (toggle == 04)
      u8g2.drawBitmap(8, 0, 14, 62, dino_022);  // dino 112*62
    else if (toggle == 05)
      u8g2.drawBitmap(8, 0, 14, 62, dino_023);  // dino 112*62
    else if (toggle == 06)
      u8g2.drawBitmap(8, 0, 14, 62, dino_024);  // dino 112*62
    else
      u8g2.drawBitmap(8, 0, 14, 62, dino_025);  // dino 112*62
  } while (u8g2.nextPage());
  toggle++;
  delay(100);
  Serial.print("Toggle = ");
  Serial.println(toggle);
  if (toggle > 7) toggle = 0;
}
